    <!-- JavaScript for Interactivity -->
    <script>
        // State
        let currentView = 'home';
        let isMobileMenuOpen = false;

        // Elements
        const homeView = document.getElementById('home-view');
        const innerView = document.getElementById('inner-view');
        const mobileMenuOverlay = document.getElementById('mobile-menu-overlay');
        const body = document.body;

        // View Switcher Function
        function switchView(viewName) {
            currentView = viewName;
            
            // Scroll to top
            window.scrollTo({ top: 0, behavior: 'smooth' });

            if (viewName === 'home') {
                innerView.classList.add('hidden');
                homeView.classList.remove('hidden');
                homeView.classList.add('fade-in');
            } else if (viewName === 'inner') {
                homeView.classList.add('hidden');
                innerView.classList.remove('hidden');
                innerView.classList.add('fade-in');
            }
        }

        // Mobile Menu Toggle
        function toggleMobileMenu() {
            isMobileMenuOpen = !isMobileMenuOpen;
            if (isMobileMenuOpen) {
                mobileMenuOverlay.classList.remove('translate-x-full');
                mobileMenuOverlay.classList.add('translate-x-0');
                body.style.overflow = 'hidden'; // Prevent scrolling bg
            } else {
                mobileMenuOverlay.classList.remove('translate-x-0');
                mobileMenuOverlay.classList.add('translate-x-full');
                body.style.overflow = 'auto';
            }
        }

        // Accordion Toggle
        function toggleAccordion(id) {
            const content = document.getElementById(id);
            const icon = document.getElementById('icon-' + id);
            
            // Check if open
            if (content.classList.contains('open')) {
                content.classList.remove('open');
                icon.textContent = 'add';
                icon.classList.remove('text-ikt-blue');
            } else {
                // Close others (optional, but requested behavior implies accordion 'focus')
                // For this demo, we allow multiple open or just one. Let's do one for cleaner UI.
                document.querySelectorAll('.accordion-content').forEach(el => {
                   if (el.id !== id) {
                       el.classList.remove('open');
                       // Reset icons for others
                       const otherIcon = document.getElementById('icon-' + el.id);
                       if(otherIcon) otherIcon.textContent = 'add';
                   }
                });

                content.classList.add('open');
                icon.textContent = 'remove';
                icon.classList.add('text-ikt-blue');
            }
        }

        // Init: Ensure correct state
        document.addEventListener('DOMContentLoaded', () => {
            // Check URL hash if we want deep linking simulation, or just default
            // Default is home
        });

    </script>
